import cv2 import time from vidgear.gears import CamGear stream = CamGear(source='https://www.youtube.com/watch?v=Jli7Ft4DUDQ', stream_mode = True, logging=True).start() # YouTube Video URL as input while True: frame = stream.read() frame = cv2.line(frame, (20,200), (200,20), (0, 0, 255), 10, 10) if frame is None: break cv2.namedWindow("Frame", cv2.WINDOW_NORMAL) cv2.resizeWindow("Frame", 400, 300) cv2.imshow("Frame", frame) key = cv2.waitKey(1) & 0xFF if key == ord("q"): break cv2.destroyAllWindows() stream.stop()
